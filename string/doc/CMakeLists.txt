include(Ryppl)
include(RypplDocumentation)
include(RypplDoxygen)

ryppl_find_and_use_package(Boostbook)

set (StringAlgoHeaders
  ../../include/boost/algorithm/string.hpp
  ../../include/boost/algorithm/string_regex.hpp
  ../../include/boost/algorithm/string
)

set(doxy_xml ${CMAKE_CURRENT_BINARY_DIR}/string_algo_doxy.xml)
ryppl_doxygen(string_algo_doxy
  XML
  INPUT
    ${StringAlgoHeaders}
  OUTPUT
    ${doxy_xml}
)

# convert from doxygen XML to to boostbook XML
xslt_doxy_to_boostbook(
  INPUT
    ${doxy_xml}
  OUTPUT
    ${CMAKE_CURRENT_BINARY_DIR}/autodoc.xml
)


set(dbk_file ${CMAKE_CURRENT_BINARY_DIR}/string_algo.dbk)

xslt_boostbook_to_docbook(
  INPUT
    ${CMAKE_CURRENT_SOURCE_DIR}/string_algo.xml
  OUTPUT
    ${dbk_file}
  DEPENDS
    ${CMAKE_CURRENT_BINARY_DIR}/autodoc.xml
  PATH
    ${CMAKE_CURRENT_SOURCE_DIR}
    ${CMAKE_CURRENT_BINARY_DIR}
)

set(HtmlParameters
  boost.root=../../../..
  chapter.autolabel=0
  chunk.section.depth=8
  toc.section.depth=2
  toc.max.depth=2
  generate.section.toc.level=1
)


set(html_dir "${CMAKE_CURRENT_BINARY_DIR}/html")
set(manifest_file ${html_dir}/string_algo_HTML.manifest)

xslt_docbook_to_html(
  INPUT
    ${dbk_file}
  OUTPUT
    ${html_dir}
  PARAMETERS
    ${HtmlParameters}
  MANIFEST
    ${manifest_file}
)

add_to_doc(${manifest_file} html)
